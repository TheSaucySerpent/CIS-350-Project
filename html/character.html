120    def move(self, screen_width, screen_height, keys, extra_speed, is_invulnerable=False):
121        """
122        Method used for normal movement as well as dodge. Moves the given Character by their speed
123
124        Args:
125        screen_width (int) : The width of the screen
126        screen_height (int) : The height of the screen
127        keys (arr): The key pressed determines the direction the speed is added to
128        extra_speed (int): Gets added to the character's base speed, used for dodge ability
129        is_invulnerable (bool, optional): Used to make the player invulnerable during dodge
130        """
131        new_x = self.x
132        new_y = self.y
133
134        if is_invulnerable:
135            self.invulnerable = True
136
137        direction = None  # Store the current movement direction
138
139        if keys[pygame.K_w]:
140            new_y -= self.speed + extra_speed
141            # direction = 'up'
142        if keys[pygame.K_a]:
143            new_x -= self.speed + extra_speed
144            direction = 'left'
145        if keys[pygame.K_s]:
146            new_y += self.speed + extra_speed
147            # direction = 'down'
148        if keys[pygame.K_d]:
149            new_x += self.speed + extra_speed
150            direction = 'right'
151
152        if direction:
153            self.set_image(direction)  # Set the character's image based on the current direction
154
155        # Keeps character within bounds
156        if 0 <= new_x <= screen_width - self.width and 0 <= new_y <= screen_height - self.height:
157            self.x = new_x
158            self.y = new_y
159
160        self.invulnerable = False

241    def pick_up(self, current_room):
242        """
243        Allow the character to pick up items from the current room.
244        This method sees if the character pressed e on an item, and, if so, adds it to inventory.
245
246        Args:
247        current_room (Room): The current room the character is in.
248        """
249        keys = pygame.key.get_pressed()
250        items = current_room.items.copy()  # Make a copy of the items in the room to avoid modifying the original
251        # list while iterating
252        for item in items:
253            if (self.x < item.x + item.width and
254                    self.x + self.width > item.x and
255                    self.y < item.y + item.height and
256                    self.y + self.height > item.y) and keys[pygame.K_e]:
257                if type(item) == weapons.Weapon or type(item) == weapons.Shotgun:
258                    self.gun = item
259                    current_room.items.remove(item)
260                else:
261                    # Add the object to the character's inventory
262                    self.inventory.append(item)
263                    # Remove the object from the list of objects in the room
264                    current_room.items.remove(item)
